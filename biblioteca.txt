create database biblioteca;

use biblioteca;

CREATE TABLE livro (
codLivro int PRIMARY KEY auto_increment,
genero varchar(30),
editora varchar(30),
autor varchar(80),
titulo varchar(50)
);

CREATE TABLE cliente (
ra int PRIMARY KEY auto_increment,
dataNascimento date,
nome varchar(100),
turma varchar(95),
email varchar(50) unique,
telefone varchar(15)
);

CREATE TABLE bibliotecario (
codBibliotecario int PRIMARY KEY auto_increment,
nome varchar(50),
email varchar(50) unique,
dataNascimento date,
cpf varchar(20) unique,
telefone varchar(15),
rg varchar(20) unique,
logradouro varchar(30),
num varchar(5),
bairro varchar(30),
cidade varchar(30),
uf varchar(2),
complemento varchar(20),
cep varchar(15)
);

CREATE TABLE emprestimo_devolucao (
codEmprestimo int PRIMARY KEY auto_increment,
data_emprestimo varchar(15),
data_devolucao varchar(15),
ra int,
codBibliotecario int,
codLivro int,
FOREIGN KEY(ra) REFERENCES cliente (ra),
FOREIGN KEY(codBibliotecario) REFERENCES bibliotecario (codBibliotecario),
FOREIGN KEY(codLivro) REFERENCES livro (codLivro)
);
